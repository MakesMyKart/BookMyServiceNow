const auth = firebase.auth(); // Sign Up User function signUp() { var email = document.getElementById("email").value; var password = document.getElementById("password").value; auth.createUserWithEmailAndPassword(email, password) .then((userCredential) => { alert("Sign up successful!"); }) .catch((error) => { alert(error.message); }); } // Sign In User function signIn() { var email = document.getElementById("email").value; var password = document.getElementById("password").value; auth.signInWithEmailAndPassword(email, password) .then((userCredential) => { alert("Login successful!"); }) .catch((error) => { alert(error.message); }); }
